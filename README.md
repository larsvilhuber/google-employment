```r
google_data <- rbind(google_employment,
                  data.frame(Year=seq(max(google_employment$Year)+1,max(google_employment$Year)+20),
                             Employment=NA)
                  )
model <- lm(google_data$Employment ~ google_data$Year)
summary(model)
```
